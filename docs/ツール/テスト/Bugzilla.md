    """ãƒã‚°ä½œæˆ"""

        "description": description,
        "version": "1.0",
        "severity": "normal",
        "priority": "P2"
    }

    """ãƒã‚°æ›´æ–°"""

        "Authorization": f"Bearer {API_KEY}",

    data = {}
    if status:
        data["status"] = status
    if comment:

    response = requests.get(url, headers=headers, params=params)
    return response.json()
